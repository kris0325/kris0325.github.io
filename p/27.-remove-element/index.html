class Solution11 {
    public int removeElement(int[] nums, int val) {
        int left = 0;
        int right =nums.length -1;
        int tmp = 0;
        while (left <= right) {
            if(nums[left] == val){
                tmp = nums[left];
                nums[left] = nums[right];
                nums[right] = tmp;
                right--;
            } else {
                left++;
            }
        }
        return left;
    }
}
