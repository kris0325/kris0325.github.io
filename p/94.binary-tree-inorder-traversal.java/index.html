 * 
 *

class Solution1 {
    List<Integer> result = new ArrayList<>();
    public List<Integer> inorderTraversal(TreeNode root) {
        
        if (null == root) {

            return result;
        }
        if(root.left != null){
            inorderTraversal(root.left);
        }
        result.add(root.val);
        if(root.right != null){
            inorderTraversal(root.right);
        }
        return result;
    }
    
}

class Solution {
    List<Integer> result = new ArrayList<>();
    public List<Integer> inorderTraversal(TreeNode root) {
        Stack<TreeNode> stack = new Stack<>();
        TreeNode cur = root;
        while(cur != null || !stack.isEmpty()){
            while(cur!= null){
                stack.push(cur);
                cur = cur.left;
            }
            cur = stack.pop();
            result.add(cur.val);
            cur = cur.right;
        }
        return result;
    }
    
}
