// @lc code=start

class ListNode {
    int val;
    ListNode next;
    ListNode(){};
    ListNode(int val){
        this.val = val;
    };
}

class MyLinkedList {

    int size;
    ListNode head;


    public MyLinkedList() {
        size = 0;
        head = new ListNode(0);
    }
    
    public int get(int index) {
        if(index < 0 || index >= size){
            return -1;
        }
        ListNode cur = head;
        for(int i = 0; i <= index; i++){
            cur = cur.next;
        }
        return cur.val;
    }
    
    public void addAtHead(int val) {
        addAtIndex(0, val);
    }
    
    public void addAtTail(int val) {
        addAtIndex(size, val);

    }
    
    
    public void addAtIndex(int index, int val) {
        if(index > size){
            return;
        }
        if(index < 0){
            index = 0; 
        }
        size++;
        //å¯»æ‰¾åˆ°è¦æ’å…¥èŠ‚ç‚¹çš„å‰é©±
        ListNode pre = head;
        for(int i = 0; i<index; i++){
            pre = pre.next;
        }
        ListNode target = new ListNode(val);
        target.next = pre.next;
        pre.next = target;
    }
    
    //åˆ é™¤ç¬¬indexèŠ‚ç‚¹
    public void deleteAtIndex(int index) {
        if(index >= size || index <0){
            return;
        }
        size--;
        ListNode pre = head;
        for(int i =0; i<index;i++ ){
            pre = pre.next;
        }
        pre.next = pre.next.next;

    }
}
