    public int countOfAirLanes(List<Interval> airplanes) {
        int maxCount = 0;
        int currentCount = 0;
        List<Event> allEvents = new ArrayList<>();
        for (Interval interval : airplanes) {
            //Ëµ∑È£û‰∫ã‰ª∂
            allEvents.add(new Event(interval.start, true));
            //ÈôçËêΩ‰∫ã‰ª∂
            allEvents.add(new Event(interval.end, false));
        }
        //ÂØπ‰∫ã‰ª∂ÊåâÊó∂Èó¥ËÉúÂ∫èÊéíÂ∫èÔºöÊó∂Èó¥Áõ∏ÂêåÁöÑÊÉÖÂÜµÔºåÈôçËêΩ‰∫ã‰ª∂‰ºòÂÖà
        Collections.sort(allEvents,(a,b) -> {
            if (a.time == b.time) {
                return a.isStart ? 1 : -1;
            }
            return a.time - b.time;
        });

        for (Event event : allEvents) {
            //Ëµ∑È£ûÔºåcurrentCountÂä†1
            if (event.isStart) {

        }
        return maxCount;
    }
