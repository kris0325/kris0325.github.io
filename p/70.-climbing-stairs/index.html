class Solution {
    public int climbStairs(int n) {
        if (n <= 2) {
            return n;
        }
        int[] dp = new int[n + 1];
        dp[1] = 1;
        dp[2] = 2;
        //Ê≥®ÊÑè‰∏ãÊ†áindexÂºÄÂßã‰∫éi=3
        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i - 2] + dp[i - 1];
            // System.out.println("dp[i]"+ "i:"+ i+":"+dp[i]);
        }
        return dp[n];
    }
}
