   class Solution {
        public int rob(int[] nums) {
            return stealStore(nums);
        }

        public int stealStore(int[] stores) {
            if (stores == null || stores.length == 0) {
                return 0;
            }
            if (stores.length == 1) {
                return stores[0];
            }
            int[] dp = new int[stores.length];
            dp[0] = stores[0];
            dp[1] = Math.max(dp[0], stores[1]);
            for (int i = 2; i < stores.length; i++) {
