class Solution {
    public int search(int[] nums, int target) {
        int left = 0;
        int right = nums.length -1;
        while (left <= right) {
            int middle = left + (right -left)/2;
            //targetÂú®Â∑¶Âå∫Èó¥
            if(target < nums[middle]){
                //middleÊåáÈíà‰∏ãÊ†áÈúÄË¶ÅÂêëÂ∑¶ËæπÁßªÂä®Êó∂ÔºåÈÅøÂÖç2‰∏™ÂÖÉÁ¥†Êï∞ÁªÑÁöÑcase‰∏ãÊ†á‰∏çÁßªÂä®ÔºåÊâÄ‰ª•ÈúÄË¶Å-1
                right = middle -1;
                
            } else if(target > nums[middle]){
                //targetÂú®Âè≥Âå∫Èó¥
                //ÂêåÁêÜÔºåmiddleÊåáÈíàÂêëÂè≥ÁßªÂä®Êó∂Ôºå‰∏ãÊ†áÈúÄË¶Å+1Ôºå
                left = middle +1;
            } else{
                return middle;
            }            
        }

        return -1;
    }
}
