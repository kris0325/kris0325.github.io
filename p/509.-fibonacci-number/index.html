class Solution {
    public int fib(int n) {
        //æ³¨æ„å¤„ç†è¾¹ç•Œæ¡ä»¶
        if(n<2) return n;
        int[] fib = new int[n+1];
        fib[0]= 0;
        fib[1]= 1;
        for(int i =2; i <=n ;i++){
            fib[i] = fib[i-1]+fib[i-2];
            // System.out.println("fib[n]:"+fib[n]);
        }
        return fib[n];
    }
}
