 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;

 * }

class Solution {
    public ListNode removeElements(ListNode head, int val) {
        // Â§¥ËäÇÁÇπ ÊòØÊª°Ë∂≥ Node.val == val ÁöÑËäÇÁÇπÔºåÂà†Èô§Â§¥ËäÇÁÇπ,
        // ‰ΩÜÊòØnextËäÇÁÇπÁöÑÂÄºÂèØËÉΩ‰πüÁõ∏Á≠âÔºåÊâÄ‰ª•Âæ™ÁéØÈÅçÂéÜ
        while(head != null && head.val == val){
            head = head.next;
        }
        if (head == null) {
            return head;
        }

        ListNode cur = head; 
        while (cur.next != null) {
            if(cur.next.val == val){
                cur.next = cur.next.next;
            } else {
                cur = cur.next;
            }
        }
        return head;

    }
}
